56        * @param direction Direction key from 0 to 8 (inclusive)

59       public void moveBlob(int direction, int speed) {
60           /* 
61           1   2   3 
62           8   0   4 
63           7   6   5 
64           */
65   
66           if (direction < 0 || direction > 8 || speed < 1)
67               throw new IllegalArgumentException("Illegal direction or speed values whilst trying to move Blob.");
68   
69           for (int i = 0; i < speed; i++) {
70               switch (direction) {
71                   case 0:
72                       break;
73                   case 1:
74                       posX--; posY++;
75                       break;
76                   case 2:
77                               posY++;
78                       break;
79                   case 3:
80                       posX++; posY++;
81                       break;
82                   case 4:
83                       posX++;
84                       break;
85                   case 5:
86                       posX++; posY--;
87                       break;
88                   case 6:
89                               posY--;
90                       break;
91                   case 7:
92                       posX--; posY--;
93                       break;
94                   case 8:
95                       posX--;
96                       break;
97               }
98           }
99   
100          if (direction != 0) {
101              if (size < 25) modEnergy(-3);
102              else modEnergy((int) Math.round(-0.12 * size));
103  
104              modEnergy(Math.round(-agro / 30));
105          }
106  
107          correctPos();
108          if (foodX == null) prefDir = direction;
109          else prefDir = 0;
110          tempDir = direction;
111      }

113      /**
114       * The Blob will move randomly with maximum speed if it has not found any food nearby.
115       * If its exclusion key is greater than 0, movement is restricted per the exclusion procedure.
116       * The Blob will generally prefer to walk to the same direction it was walking before.
117       * Else if it has spotted a food source, it will move with appropriate speed to that food source.
118       */
119      public void decideMovement() {
120          int exKey = getExKey();
121          int freeDir = -1;
122          int exDir = -1;
123  
124          setReverseDirectionKey();
125  
126          if (exKey == 0) {
127              do {
128                  freeDir = rng.nextInt(25);
129              } while (freeDir == reverseDir);
130          } else {
131              do {
132                  exDir = getValidDirKey(exKey);
133              } while (exDir == reverseDir);
134          }
135  
136          if (foodX != null) {
137              moveBlob(dirFood, adjustedSpeed);
138          }
139          else if (exKey == 0 && prefDir == 0) {
140              moveBlob(freeMovement[freeDir], speed);
141          }
142          else if (exKey == 0 && prefDir > 0) {
143              if (calcProb(dirProb)) moveBlob(prefDir, speed);
144              else moveBlob(freeMovement[freeDir], speed);
145          }
146          else if (exKey > 0 && prefDir == 0) {
147              moveBlob(exDir, speed);
148          }
149          else if (exKey > 0 && prefDir > 0) {
150              boolean prefDirAvailable = false;
151              int[] dirCheck = getValidDirArray(exKey);
152  
153              for (int j : dirCheck) {
154                  if (j == prefDir) {
155                      prefDirAvailable = true;
156                      break;
157                  }
158              }
159  
160              if (prefDirAvailable && calcProb(dirProb)) moveBlob(prefDir, speed);
161              else moveBlob(exDir, speed);
162          }
163          setReverseDirectionKey();
164      }

173      /**
174       * Scans the area for a food source and marks its position if it has been found.
175       * "Area" is defined by the sense property of the Blob. The Blob will try to
176       * choose the nearest food source possible.
177       * @return True if nearby food has been located.
178       */
179      public boolean senseFood() {
180          int foodX, foodY;
181          int fillIndex = -1;
182          Food[] availableFood = new Food[10];
183  
184          for (Map.Entry<Integer, Food> foodEntity : foodHash.entrySet()) {
185              foodX = foodEntity.getValue().posX;
186              foodY = foodEntity.getValue().posY;
187  
188              if (foodX - sense <= posX && foodX + sense >= posX &&
189                  foodY - sense <= posY && foodY + sense >= posY) {
190                  fillIndex++;
191                  availableFood[fillIndex] = foodEntity.getValue();
192              } if (fillIndex == 9) break;
193          }
194  
195          if (fillIndex > 0) {
196              int tempX = Math.abs(availableFood[0].posX - posX);
197              int tempY = Math.abs(availableFood[0].posY - posY);
198              int compX, compY;
199              int index = 0;
200  
201              for (int i = 0; i < fillIndex; i++) {
202                  compX = Math.abs(availableFood[(i + 1)].posX - posX);
203                  compY = Math.abs(availableFood[(i + 1)].posY - posY);
204                  if ((compX < tempX && compY <= tempY) ||
205                      (compX <= tempX && compY < tempY)) {
206                      tempX = compX;
207                      tempY = compY;
208                      index = i + 1;
209                  }
210              }
211  
212              this.foodX = availableFood[index].posX;
213              this.foodY = availableFood[index].posY;
214              return true;
215          } else if (fillIndex == 0) {
216              this.foodX = availableFood[0].posX;
217              this.foodY = availableFood[0].posY;
218              return true;
219          }
220  
221          this.foodX = null;
222          this.foodY = null;
223          return false;
224      }

228       * If Blob is on a block with food on it, the method returns true.
229       * @return True if food has been eaten
230       */
231      public boolean eatFood() {
232          boolean ateFood = false;
233          if (foodX == null || foodY == null) return false;
234  
235          for (Map.Entry<Integer, Food> foodEntity : foodHash.entrySet()) {
236              if (foodEntity.getValue().available &&
237                      foodEntity.getValue().posX == foodX &&
238                      foodEntity.getValue().posY == foodY &&
239                      posX == foodX &&
240                      posY == foodY) {
241                  
242                  modEnergy(foodEntity.getValue().satiety);
243                  foodAmount--;
244                  foodEaten++;
245                  ateFood = true;
246  
247                  if (printEnvEnabled) {
248                      removeFoodQueue.add(foodEntity.getKey());
249                      foodQueueSize++;
250                      foodEntity.getValue().available = false;
251                  } else foodHash.remove(foodEntity.getKey());
252  
253                  if (energy >= 100) {
254                      setEnergy(100);
255                      return true;
256                  }
257              }
258          }
259          adjustedSpeed = speed;
260          return ateFood;

263      /**
264       * If the Blob meets another Blob on the same block, it will eat it if
265       * it's 20% bigger than it. A Blob can eat 1 other Blob at most in one go.
266       * @return True if a Blob has been eaten
267       */
268      public boolean eatBlob() {
269          if (size < 8) return false;
270          Blob blob;
271          int blobsEaten = 0;
272  
273          for (Map.Entry<Integer, Blob> blobEntity : blobHash.entrySet()) {
274              blob = blobEntity.getValue();
275  
276              if (energy >= 100 && blobsEaten > 0) return true;
277              if (energy >= 100 && blobsEaten == 0) return false;
278  
279              if (blob.posX == posX && blob.posY == posY && blobsEaten <= 1 &&
280                      blob.age >= 10 && size >= blob.size + (blob.size * 0.2)) {
281  
282                  int minSizeDiff = (int) (size - (blob.size + (blob.size * 0.2)));
283                  int agroDiff = agro - blob.agro;
284                  int eatProb = 50 + agroDiff;
285  
286                  if (calcProb((eatProb + minSizeDiff))) {
287                      modEnergy((blob.size / 8));
288                      modSize(1);
289                      blobHash.remove(blobEntity.getKey());
290                      blobDeaths++;
291                      blobsEaten++;
292                      blobAmount--;
293                      Log.blobsEaten++;
294                  } else {
295                      modEnergy(-(blob.size / 6));
296                      blob.modEnergy(-minSizeDiff);
297                      if (rng.nextBoolean()) blob.modAgro(1);
298                      blobsDefeated++;
299                  }
300              }
301          }
302          growOrShrink();
303          return blobsEaten > 0;
304      }

326          7   6   5 
327          */
328  
329          if (foodX == null || foodY == null) return 0;
330  
331          int tempX, tempY;
332          tempX = foodX - posX;
333          tempY = foodY - posY;
334  
335          if      (tempX == 0 && tempY == 0)  dirFood = 0;
336          else if (tempX < 0 && tempY > 0)    dirFood = 1;
337          else if (tempX == 0 && tempY > 0)   dirFood = 2;
338          else if (tempX > 0 && tempY > 0)    dirFood = 3;
339          else if (tempX > 0 && tempY == 0)   dirFood = 4;
340          else if (tempX > 0 && tempY < 0)    dirFood = 5;
341          else if (tempX == 0 && tempY < 0)   dirFood = 6;
342          else if (tempX < 0 && tempY < 0)    dirFood = 7;
343          else if (tempX < 0 && tempY == 0)   dirFood = 8;
344  
345          tempX = Math.abs(tempX);
346          tempY = Math.abs(tempY);
347          if (speed == 1) adjustedSpeed = 1;
348          else calcAdjustedSpeed(tempX, tempY);
349          return dirFood;
350      }

353       * Calculates the adjusted speed value by checking where food is located.
354       * @param tempX The absolute x value determined by calcDirectionKey()
355       * @param tempY The absolute y value determined by calcDirectionKey()
356       * @return The adjusted speed
357       */
358      public int calcAdjustedSpeed(int tempX, int tempY) {
359          if (tempX >= tempY) {
360              if (tempX <= speed && tempX > 0) adjustedSpeed = tempX;
361              else adjustedSpeed = speed;
362          } else {
363              if (tempY <= speed && tempY > 0) adjustedSpeed = tempY;
364              else adjustedSpeed = speed;
365          }
366          return adjustedSpeed;
367      }

30       /**
31        * Gets a valid direction key. Exceptions are ruled out with the exclusion key.
32        *
33        * If a Blob is at an edge or corner of the environment, it must be prohibited from
34        * moving outside that environment. This is done by excluding direction keys that
35        * lead outside the environment. The exclusion key equals the innermost value of
36        * the excluded direction keys.
37        *
38        * E.g. exTop excludes 1 2 3 and 2 is the innermost/middle value of these, so
39        * that's the exclusion key for Top exceptions.
40        *
41        * This method randomly chooses an index and returns the valid direction key at
42        * that index of the corresponding array that is determined by the exclusion key.
43        * @param exKey The exclusion key
44        * @return A valid direction key
45        */
46       public static int getValidDirKey(int exKey) {
47           /* 
48           1   2   3 
49           8   0   4 
50           7   6   5 
51           */
52   
53           int index = 0;
54   
55           //Check if index range is 4 or 6.
56           //Odd exclusion keys exclude 5 of 9 direction keys.
57           //Even exclusion keys exclude 3 of 9 direction keys.
58           if (exKey % 2 == 1) {
59               index = rng.nextInt(12);
60           } else if (exKey % 2 == 0) {
61               index = rng.nextInt(20);
62           }
63   
64           switch (exKey) {
65               case 1:
66                   return exTopLeft[index];
67               case 2:
68                   return exTop[index];
69               case 3:
70                   return exTopRight[index];
71               case 4:
72                   return exRight[index];
73               case 5:
74                   return exBottomRight[index];
75               case 6:
76                   return exBottom[index];
77               case 7:
78                   return exBottomLeft[index];
79               case 8:
80                   return exLeft[index];
81           }
82   
83           throw new IndexOutOfBoundsException("Calc: Exclusion key is out of range.");
84       }

87           switch (exKey) {

89                   return exTopLeft;
90               case 2:
91                   return exTop;
92               case 3:
93                   return exTopRight;
94               case 4:
95                   return exRight;
96               case 5:
97                   return exBottomRight;
98               case 6:
99                   return exBottom;
100              case 7:
101                  return exBottomLeft;
102              case 8:
103                  return exLeft;
104          }

133       * Example: If the percentage parameter is 25, there is a 1/4 chance of returning true.
134       * @param percent Percentage probability of returning true
135       * @return True if random number was in the given range
